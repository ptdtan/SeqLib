14 template<typename IterType, typename BucketFunctor>
15 void bucketSort(IterType start, IterType end, BucketFunctor func)
16 {
17  // Typedef the base value of the iterator
18  typedef typename std::iterator_traits<IterType>::value_type base_value;
19  typedef std::list<base_value> base_list;
20  typedef typename base_list::iterator list_iterator;
21 
22  // Get the number of buckets needed
23  int numBuckets = func.getNumBuckets();
24 
25  // Allocate an array of lists
26  base_list* buckets = new base_list[numBuckets];
27 
28  // Place the items into buckets
29  for(IterType iter = start; iter != end; ++iter)
30  {
31  int bucket_id = func(*iter);
32  assert(bucket_id < numBuckets);
33  buckets[bucket_id].push_back(*iter);
34  }
35 
36  // Place the elements back into the array
37  IterType curr = start;
38  for(int i = 0; i < numBuckets; ++i)
39  {
40  IterType bucket_start = curr;
41  size_t count = 0;
42  for(list_iterator bucket_iter = buckets[i].begin(); bucket_iter != buckets[i].end(); ++bucket_iter)
43  {
44  ++count;
45  *curr++ = *bucket_iter;
46  }
47  std::sort(bucket_start, curr, func);
48  }
49  assert(curr == end);
50 
51  // delete lists
52  delete [] buckets;
53 }

55 template<typename T, typename PrimarySorter, typename FinalSorter>
56 void histogramSort(T* x, size_t n, int depth, PrimarySorter& primarySorter, const FinalSorter& finalSorter)
57 {
58  // Get the number of buckets needed
59  int numBuckets = primarySorter.getNumBuckets();
60 
61  // Set the functor's offset
62  primarySorter.setBucketDepth(depth);
63 
64  // Allocate an array for the bucket start points and the bucket end points
65  // TODO: Could remove at least one of these arrays
66  size_t* bucket_starts = new size_t[numBuckets];
67  size_t* bucket_ends = new size_t[numBuckets];
68  size_t* bucket_counts = new size_t[numBuckets];
69 
70  for(int i = 0; i < numBuckets; ++i)
71  {
72  bucket_starts[i] = 0;
73  bucket_ends[i] = 0;
74  bucket_counts[i] = 0;
75  }
76 
77  // Place the items into buckets
78  for(size_t i = 0; i < n; ++i)
79  {
80  int bucket_id = primarySorter.getBucket(x[i]);
81  assert(bucket_id < numBuckets);
82  bucket_counts[bucket_id]++;
83  }
84 
85  // Set up the start/end points
86  size_t sum = 0;
87  for(int i = 0; i < numBuckets; ++i)
88  {
89  size_t val = bucket_counts[i];
90  bucket_starts[i] = sum;
91  sum += val;
92  bucket_ends[i] = sum;
93  //std::cout << "Bucket count: " << bucket_counts[i] << " start: " << bucket_starts[i] << " end: " << bucket_ends[i] << "\n";
94 
95  }
96 
97  // Now, consider the bucket_starts positions to be the next (potentially) unsorted position
98  // for each bucket. Iterate through these values cycles the elements into place
99  int curr_bucket = 0; // Start on the first bucket that has element
100  while(bucket_starts[curr_bucket] == bucket_ends[curr_bucket])
101  ++curr_bucket;
102 
103  while(curr_bucket != numBuckets)
104  {
105  //std::cout << "Curr Bucket: " << curr_bucket << "\n";
106  //std::cout << "CurrIDX: " << bucket_starts[curr_bucket] << "\n";
107  int elem_offset = bucket_starts[curr_bucket];
108 
109  // Get the bucket for this element
110  int bucket_id = primarySorter.getBucket(x[elem_offset]);
111 
112  // if the element is in the correct bucket, just advance the pointer
113  if(bucket_id == curr_bucket)
114  bucket_starts[curr_bucket]++;
115  else
116  {
117  // Cycle the elements until we land back in this bucket, at which point
118  // we have sorted one position further in the current bucket
119 
120  // Hold the info for the element that has been displaced
121  T displaced = x[elem_offset];
122  bool stop = false;
123 
124  while(!stop)
125  {
126  // Get the offset of the position this element should be in
127  int displaced_bucket = primarySorter.getBucket(displaced);
128 
129  if(displaced_bucket == curr_bucket)
130  {
131  // The displaced element belongs back in the original bucket, place it there
132  x[elem_offset] = displaced;
133  bucket_starts[curr_bucket]++;
134  stop = true;
135  }
136  else
137  {
138  int cycle_offset = bucket_starts[displaced_bucket];
139 
140  // Make space for the incoming element
141  T swap = x[cycle_offset];
142  x[cycle_offset] = displaced;
143  displaced = swap;
144  bucket_starts[displaced_bucket]++;
145  }
146  }
147  }
148 
149  // At this point, the bucket_start for current bucket
150  // has moved 1 element
151  // Other buckets may have advanced as well
152  //
153  while(curr_bucket < numBuckets && bucket_starts[curr_bucket] == bucket_ends[curr_bucket])
154  ++curr_bucket;
155  }
156 
157  // Recompute the start/end points
158  sum = 0;
159  for(int i = 0; i < numBuckets; ++i)
160  {
161  size_t val = bucket_counts[i];
162  bucket_starts[i] = sum;
163  sum += val;
164  bucket_ends[i] = sum;
165  }
166 
167  // Finally, sort each bucket
168  for(int i = 0; i < numBuckets; ++i)
169  {
170  T* curr_bucket = x + bucket_starts[i];
171  size_t curr_count = bucket_counts[i];
172 
173  if(curr_count < 2)
174  continue;
175 
176  assert(curr_count == (bucket_ends[i] - bucket_starts[i]));
177 
178  //std::cout << "Sorting bucket " << i << "\n";
179  const int max_depth = 100;
180  const size_t min_elements = 1000;
181 
182  // terminate the bucket sort under 3 conditions:
183  // 1) the bucket is degenerate (no more histogram sorting can be done) - this comes from the functor
184  // 2) the number of elements in the bucket is low
185  // 3) the max depth has been hit
186  if(primarySorter.isBucketDegenerate(i))
187  {
188  std::sort(curr_bucket, curr_bucket + curr_count, finalSorter);
189  }
190  else if(bucket_counts[i] < min_elements || depth >= max_depth)
191  {
192  int next_char = (depth + 1) * primarySorter.getBucketLen();
193  mkqs2(curr_bucket, curr_count, next_char, primarySorter, finalSorter);
194  //std::sort(curr_bucket, curr_bucket + bucket_counts[i], finalSorter);
195  }
196  else
197  {
198  histogramSort(curr_bucket, curr_count, depth + 1, primarySorter, finalSorter);
199  }
200  }
201 
202  // delete arrays
203  delete [] bucket_counts;
204  delete [] bucket_starts;
205  delete [] bucket_ends;
206 }

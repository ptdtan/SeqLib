80 {
81  int w;
82  uint64_t x;
83  int64_t l, y = 0;
84  x = itr->p[0] << (64 - itr->r) | (itr->p != itr->stail && itr->r != 64? itr->p[1] >> itr->r : 0);
85  if (x>>63 == 0) {
86  if ((w = 0x333333335555779bll>>(x>>59<<2)&0xf) == 0xb && x>>58 == 0) return 0;
87  l = (x >> (64 - w)) - 1;
88  y = x << w >> (64 - l) | 1u << l;
89  w += l;
90  } else w = y = 1;
91  *c = x << w >> (64 - e->abits);
92  w += e->abits;
93  if (itr->r > w) itr->r -= w;
94  else ++itr->p, itr->r = 64 + itr->r - w;
95  return y;
96 }

99 {
100  int64_t l = rld_dec0(e, itr, _c);
101  if (l == 0 || *_c > e->asize) {
102  uint64_t last = rld_last_blk(e);
103  if (itr->p - *itr->i > RLD_LSIZE - e->ssize) {
104  if (is_free) {
105  free(*itr->i); *itr->i = 0;
106  }
107  itr->shead = *++itr->i;
108  } else itr->shead += e->ssize;
109  if (itr->shead == rld_seek_blk(e, last)) return -1;

111  itr->q = (uint8_t*)itr->p;

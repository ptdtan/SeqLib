130  // Visit each vertex in the graph and call the visit functor object
131  template<typename VF>
132  bool visit(VF& vf)
133  {
134  bool modified = false;
135  vf.previsit(this);
136  VertexPtrMapConstIter iter = m_vertices.begin();
137  for(; iter != m_vertices.end(); ++iter)
138  {
139  modified = vf.visit(this, iter->second) || modified;
140  }
141  vf.postvisit(this);
142  return modified;
143  }

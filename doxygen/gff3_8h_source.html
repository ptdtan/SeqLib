10 /* Annotation record from a GFF3 file. Attributes define in the spec (those
11  * starting with upper case letters) are parsed into fields of this
12  * object. User defined attributes (starting with lower-case characters) are
13  * stored as in a list, along with a copy of the string versions of the spec
14  * attributes. All strings stored in the object have been un-escaped.
15  * All storage for the object is allocated by the gff3File object.
16  * For discontinuous features, there are multiple gff3Ann objects.
17  * These objects are stored in a double-linked list, and all references
18  * point to the first one in ascending start order.*/

20  struct gff3Ann *prevPart; /* Discontinuous features have linked annotation */

22  char *seqid; /* The ID of the landmark used to establish the coordinate
23  * system for the current feature. IDs may contain any
24  * characters. */
25  char *source; /* The source is a free text qualifier intended to describe
26  * the algorithm or operating procedure that generated this
27  * feature. Typically this is the name of a piece of
28  * software, such as "Genescan" or a database name, such as
29  * "Genbank." In effect, the source is used to extend the
30  * feature ontology by adding a qualifier to the type
31  * creating a new composite type that is a subclass of the
32  * type in the type column. */
33 
34  char *type; /* The type of the feature (previously called the "method").
35  * This is constrained to be either: (a) a term from the
36  * "lite" sequence ontology, SOFA; or (b) a SOFA accession
37  * number. The latter alternative is distinguished using the
38  * syntax SO:000000. */
39 
40  int start; /* The start and end of the feature, in 0-based, half open
41  * integer coordinates, relative to the landmark given in
42  * seqid. Start is always less than or equal to end. For
43  * zero-length features, such as insertion sites, start equals
44  * end and the implied site is to the right of the indicated
45  * base in the direction of the landmark.*/
46  int end;
47  float score; /* The score of the feature, a floating point number. As in
48  earlier versions of the format, the semantics of the score
49  are ill-defined. It is strongly recommended that E-values
50  be used for sequence similarity features, and that
51  P-values be used for ab initio gene prediction features. */
52  boolean haveScore; /* was score specified? */

54  char *strand; /* The strand of the feature. '+' for positive strand
55  * (relative to the landmark), '-' for minus strand, and
56  * NULL for features that are not stranded. In addition,
57  * '?' can be used for features whose strandedness is
58  * relevant, but unknown. */
59 
60  int phase; /* For features of type "CDS", the phase indicates where the
61  * feature begins with reference to the reading frame. The
62  * phase is one of the integers 0, 1, or 2, indicating the
63  * number of bases that should be removed from the beginning of
64  * this feature to reach the first base of the next codon. In
65  * other words, a phase of 0 indicates that the next codon
66  * begins at the first base of the region described by the
67  * current line, a phase of 1 indicates that the next codon
68  * begins at the second base of this region, and a phase of 2
69  * indicates that the codon begins at the third base of this
70  * region. This is NOT to be confused with the frame, which is
71  * simply start modulo 3. For forward strand features, phase
72  * is counted from the start field. For reverse strand
73  * features, phase is counted from the end field. The phase is
74  * REQUIRED for all CDS features. and -1 for other features. */
75

79  * letter) are stored in the attrs list. */

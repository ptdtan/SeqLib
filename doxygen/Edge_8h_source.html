22 // Packed structure holding the direction and comp of an edge
23 // The EdgeDir/EdgeComp enums (which only have values 0/1) are used
24 // as the interface for this class so the set flags are cast to/from
25 // these types.
26 struct EdgeData
27 {
28  public:
29  EdgeData() {}
30 
31  // Setters
32  void setDir(EdgeDir dir)
33  {
34  if(dir == ED_ANTISENSE)
35  m_data.set(DIR_BIT, true);
36  else
37  m_data.set(DIR_BIT, false);
38  }
39 
40  void setComp(EdgeComp comp)
41  {
42  if(comp == EC_REVERSE)
43  m_data.set(COMP_BIT, true);
44  else
45  m_data.set(COMP_BIT, false);
46  }
47 
48  void flipDir() { m_data.flip(DIR_BIT); }
49  void flipComp() { m_data.flip(COMP_BIT); }
50 
51  // Getters
52  inline EdgeDir getDir() const
53  {
54  return m_data.test(DIR_BIT) ? ED_ANTISENSE : ED_SENSE;
55  }
56 
57  inline EdgeComp getComp() const
58  {
59  return m_data.test(COMP_BIT) ? EC_REVERSE : EC_SAME;
60  }
61 
62  private:
63  static const size_t DIR_BIT = 0;
64  static const size_t COMP_BIT = 1;
65  BitChar m_data;
66 };

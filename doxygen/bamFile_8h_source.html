67 INLINE int bamUnpackCigarElement(unsigned int packed, char *retOp)
68 /* Given an unsigned int containing a number of bases and an offset into an
69  * array of BAM-enhanced-CIGAR ASCII characters (operations), store operation
70  * char into *retOp (retOp must not be NULL) and return the number of bases. */
71 {
72 #ifdef USE_BAM
73 // decoding lifted from samtools bam.c bam_format1_core(), long may it remain stable:
74 #define BAM_DOT_C_OPCODE_STRING "MIDNSHP=X"
75 int n = packed>>BAM_CIGAR_SHIFT;
76 int opcode = packed & BAM_CIGAR_MASK;
77 if (opcode >= strlen(BAM_DOT_C_OPCODE_STRING))
78  errAbort("bamUnpackCigarElement: unrecognized opcode %d. "
79  "(I only recognize 0..%lu [" BAM_DOT_C_OPCODE_STRING "]) "
80  "Perhaps samtools bam.c's bam_format1 encoding changed? If so, update me.",
81  opcode, (unsigned long)(strlen(BAM_DOT_C_OPCODE_STRING)-1));
82 *retOp = BAM_DOT_C_OPCODE_STRING[opcode];
83 return n;
84 #else // no USE_BAM
85 errAbort(COMPILE_WITH_SAMTOOLS, "bamUnpackCigarElement");
86 return 0;
87 #endif// USE_BAM
88 }

9 struct sufaFileHeader
10 /* Short read index file binary file header. A sufa file starts with this fixed 128 byte
11  * structure. It is followed by the following sections:
12  * chromosome name strings - zero terminated. Padded with zero to 4 byte boundary
13  * chromosome sizes (32 bits each)
14  * chromosome DNA - one byte per base lower case. A zero between each chrom, and a zero before
15  * and after (to make some end conditions easier). Padded if need be with
16  * additional zeroes to 4 base boundary.
17  * suffixArray - 32 bits for each indexed base */
18  {
19  bits32 magic; /* Always SUFA_MAGIC */
20  bits16 majorVersion; /* This version changes when backward compatibility breaks. */
21  bits16 minorVersion; /* This version changes whenever a feature is added. */
22  bits64 size; /* Total size to memmap, including header. */
23  bits32 chromCount; /* Total count of chromosomes/contigs in file. */
24  bits32 chromNamesSize; /* Size of names of all contigs (including zeroes at end),
25  padded to 4 byte boundary as needed). */
26  bits64 arraySize; /* Total number of bases actually indexed (non-N, unmasked). */
27  bits64 dnaDiskSize; /* Size of DNA on disk with zero separators. Padded to 4 byte boundary */
28  bits64 reserved[11];/* All zeroes for now. */
29  };

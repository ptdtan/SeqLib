42  inline char getChar(size_t idx) const
43  {
44  // Calculate the Marker who's position is not less than idx
45  const LargeMarker& upper = getUpperMarker(idx);
46  size_t current_position = upper.getActualPosition();
47  assert(current_position >= idx);
48 
49  size_t symbol_index = upper.unitIndex;
50 
51  // Search backwards (towards 0) until idx is found
52  while(current_position > idx)
53  {
54  assert(symbol_index != 0);
55  symbol_index -= 1;
56  current_position -= m_rlString[symbol_index].getCount();
57  }
58 
59  // symbol_index is now the index of the run containing the idx symbol
60  const RLUnit& unit = m_rlString[symbol_index];
61  assert(current_position <= idx && current_position + unit.getCount() >= idx);
62  return unit.getChar();
63  }

102  // Return a LargeMarker with values that are interpolated by adding
103  // the relative count nearest to the requested position to the last
104  // LargeMarker
105  inline LargeMarker getInterpolatedMarker(size_t target_small_idx) const
106  {
107  // Calculate the position of the LargeMarker closest to the target SmallMarker
108  size_t target_position = target_small_idx << m_smallShiftValue;
109  size_t curr_large_idx = target_position >> m_largeShiftValue;
110 
111  LargeMarker absoluteMarker = m_largeMarkers[curr_large_idx];
112  const SmallMarker& relative = m_smallMarkers[target_small_idx];
113  alphacount_add16(absoluteMarker.counts, relative.counts);
114  absoluteMarker.unitIndex += relative.unitCount;

120  // Return the number of times char b appears in bwt[0, idx]
121  inline BaseCount getOcc(char b, size_t idx) const
122  {
123  // The counts in the marker are not inclusive (unlike the Occurrence class)
124  // so we increment the index by 1.
125  ++idx;
126 
127  const LargeMarker& marker = getNearestMarker(idx);
128  size_t current_position = marker.getActualPosition();
129  bool forwards = current_position < idx;
130  //printf("cp: %zu idx: %zu f: %d dist: %d\n", current_position, idx, forwards, (int)idx - (int)current_position);
131 
132  size_t running_count = marker.counts.get(b);
133  size_t symbol_index = marker.unitIndex;
134 
135  if(forwards)
136  accumulateForwards(b, running_count, symbol_index, current_position, idx);
137  else

140 /* stdargs compatibility: in a unix universe a long time ago, types of va_list
141  * were passed by value. It was assume one could do things like:
142  *
143  * va_start(args);
144  * vfprintf(fh1, fmt, args);
145  * vfprintf(fh2, fmt, args);
146  * va_end(args);
147  *
148  * and life would good. However this is not true on some modern systems (for
149  * instance gcc/glibc on x86_64), where va_args can be a pointer to some type
150  * of object). The second call to vfprintf() would then crash, since the
151  * first call modified the object that va_args was pointing to. C99 adds a
152  * va_copy macro that to address this issue. Many non-C99 system include this
153  * macro, sometimes called __va_copy. Here we ensure that va_copy is defined.
154  * If if doesn't exist, we try to define it in terms of __va_copy. If that is
155  * not available, we make the assumption that va_list can be copied by value
156  * and create our own. Our implementation is the same as used on Solaris.
157  */
158 #if defined(__va_copy) && !defined(va_copy)
159 # define va_copy __va_copy
160 #endif
161 #if !defined(va_copy)
162 # define va_copy(to, from) ((to) = (from))
163 #endif

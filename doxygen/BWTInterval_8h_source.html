17 // A BWTInterval holds a pair of integers which delineate an alignment of some string
18 // to a BWT/Suffix Array
19 struct BWTInterval
20 {
21  // Functions
22  BWTInterval() : lower(0), upper(0) {}
23  BWTInterval(int64_t l, int64_t u) : lower(l), upper(u) {}
24 
25  inline bool isValid() const { return lower <= upper; }
26  inline int64_t size() const { return upper - lower + 1; }

28  static inline bool compare(const BWTInterval& a, const BWTInterval& b)
29  {
30  if(a.lower == b.lower)
31  return a.upper < b.upper;
32  else
33  return a.lower < b.lower;
34  }

39  }

11 struct itsaFileHeader
12 /* Short read index file binary file header. A itsa file starts with this fixed 128 byte
13  * structure. It is followed by the following sections:
14  * chromosome name strings - zero terminated. Padded with zero to 4 byte boundary
15  * chromosome sizes (32 bits each)
16  * chromosome DNA - one byte per base lower case. A zero between each chrom, and a zero before
17  * and after (to make some end conditions easier). Padded if need be with
18  * additional zeroes to 4 base boundary.
19  * suffix array - 32 bits for each indexed base. Alphabetical offsets into DNA
20  * traverse array - Also 32 bits per indexed base. Helper info to traverse array like a tree.
21  * index13 - index of all 13-base prefixes in the suffix array. 4**13 32 bit values.
22  * Will have zeros in positions where there is no data for this slot.
23  * Index+1 of suffix array where there is data (to free up zero for no data).
24  * cursors13 - cursor position at each of the index13 positions*/
25  {
26  bits32 magic; /* Always ITSA_MAGIC */
27  bits16 majorVersion; /* This version changes when backward compatibility breaks. */
28  bits16 minorVersion; /* This version changes whenever a feature is added. */
29  bits64 size; /* Total size to memmap, including header. */
30  bits32 chromCount; /* Total count of chromosomes/contigs in file. */
31  bits32 chromNamesSize; /* Size of names of all contigs (including zeroes at end),
32  padded to 4 byte boundary as needed). */
33  bits64 arraySize; /* Total number of bases actually indexed (non-N, unmasked). */
34  bits64 dnaDiskSize; /* Size of DNA on disk with zero separators. Padded to 4 byte boundary */
35  bits64 reserved[11];/* All zeroes for now. */
36  };

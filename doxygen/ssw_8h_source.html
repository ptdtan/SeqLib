137 static inline uint32_t to_cigar_int (uint32_t length, char op_letter)
138 {
139  switch (op_letter) {
140  case 'M': /* alignment match (can be a sequence match or mismatch */
141  default:
142  return length << BAM_CIGAR_SHIFT;
143  case 'S': /* soft clipping (clipped sequences present in SEQ) */
144  return (length << BAM_CIGAR_SHIFT) | (4u);
145  case 'D': /* deletion from the reference */
146  return (length << BAM_CIGAR_SHIFT) | (2u);
147  case 'I': /* insertion to the reference */
148  return (length << BAM_CIGAR_SHIFT) | (1u);
149  case 'H': /* hard clipping (clipped sequences NOT present in SEQ) */
150  return (length << BAM_CIGAR_SHIFT) | (5u);
151  case 'N': /* skipped region from the reference */
152  return (length << BAM_CIGAR_SHIFT) | (3u);
153  case 'P': /* padding (silent deletion from padded reference) */
154  return (length << BAM_CIGAR_SHIFT) | (6u);
155  case '=': /* sequence match */
156  return (length << BAM_CIGAR_SHIFT) | (7u);
157  case 'X': /* sequence mismatch */
158  return (length << BAM_CIGAR_SHIFT) | (8u);
159  }
160  return (uint32_t)-1; // This never happens
161 }

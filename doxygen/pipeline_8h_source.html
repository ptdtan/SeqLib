1 /* pipeline.h - create a process pipeline that can be used for reading or
2  * writing. These pipeline objects don't go through the shell, so they
3  * avoid many of the obscure problems associated with system() and popen().
4  *
5  * Read pipelines are pipelines where a program reads output written by the
6  * pipeline, and write pipelines are where the program writes data to the
7  * pipeline. The type of pipeline is specified in the set of option flags
8  * passed to pipelineOpen(). The file at the other end of the pipeline is
9  * specified in the otherEndFile argument of pipelineOpen(), as shown here:
10  *
11  * pipelineRead:
12  *
13  * otherEndFile --> cmd[0] --> ... --> cmd[n] --> pipelineLf() etc.
14  *
15  * pipelineWrite:
16  *
17  * pipeLineFile() --> cmd[0] --> ... --> cmd[n] --> otherEndFile
18  *
19  * Specify otherEndFile as "/dev/null" for no input or no output (or to
20  * discard output). If otherEndFile is NULL, then either stdin or stdout are
21  * inherited from the current process.
22  *
23  * I/O to the pipeline is done by using the result of pipelineFd(),
24  * pipelineFile(), or pipelineLineFile().
25  *
26  * An example that reads a compressed file, sorting it numerically by the
27  * first column:
28  *
29  * static char *cmd1[] = {"gzip", "-dc", NULL};
30  * static char *cmd2[] = {"sort", "-k", "1,1n", NULL};
31  * static char **cmds[] = {cmd1, cmd2, NULL};
32  *
33  * struct pipeline *pl = pipelineOpen(cmds, pipelineRead, inFilePath, stderrFd);
34  * struct lineFile *lf = pipelineLineFile(pl);
35  * char *line;
36  *
37  * while (lineFileNext(lf, &line, NULL))
38  * {
39  * ...
40  * }
41  * pipelineWait(pl);
42  * pipelineFree(&pl);
43  *
44  * A similar example that generates data and writes a compressed file, sorting
45  * it numerically by the first column:
46  *
47  *
48  * static char *cmd1[] = {"sort", "-k", "1,1n", NULL};
49  * static char *cmd2[] = {"gzip", "-c3", NULL};
50  * static char **cmds[] = {cmd1, cmd2, NULL};
51  *
52  * struct pipeline *pl = pipelineOpen(cmds, pipelineWrite, outFilePath, stderrFd);
53  * char *line;
54  *
55  * while ((line = makeNextRow()) != NULL)
56  * fprintf(fh, "%s\n", line);
57  *
58  * pipelineWait(pl);
59  * pipelineFree(&pl);
60  *
61  * To append to an output file, use pipelineWrite|pipelineAppend:
62  *
63  * struct pipeline *pl = pipelineOpen(cmds, pipelineWrite|pipelineAppend, outFilePath, stderrFd);
64  */

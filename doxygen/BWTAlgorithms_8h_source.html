59 // Update the given interval using backwards search
60 // If the interval corrsponds to string S, it will be updated
61 // for string bS
62 inline void updateInterval(BWTInterval& interval, char b, const BWT* pBWT)
63 {
64  size_t pb = pBWT->getPC(b);
65  interval.lower = pb + pBWT->getOcc(b, interval.lower - 1);
66  interval.upper = pb + pBWT->getOcc(b, interval.upper) - 1;
67 }

69 // Update the interval pair for the right extension to symbol b.
70 // In this version the AlphaCounts for the upper and lower intervals
71 // have been calculated
72 inline void updateBothR(BWTIntervalPair& pair, char b, const BWT* pRevBWT,
73  AlphaCount64& l, AlphaCount64& u)
74 {
75  AlphaCount64 diff = u - l;
76 
77  pair.interval[0].lower = pair.interval[0].lower + diff.getLessThan(b);
78  pair.interval[0].upper = pair.interval[0].lower + diff.get(b) - 1;
79 
80  // Update the right index directly
81  size_t pb = pRevBWT->getPC(b);
82  pair.interval[1].lower = pb + l.get(b);
83  pair.interval[1].upper = pb + u.get(b) - 1;
84 }

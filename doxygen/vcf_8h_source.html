158 INLINE void vcfPrintDatum(FILE *f, const union vcfDatum datum, const enum vcfInfoType type)
159 /* Print datum to f in the format determined by type. */
160 {
161 switch (type)
162  {
163  case vcfInfoInteger:
164  fprintf(f, "%d", datum.datInt);
165  break;
166  case vcfInfoFloat:
167  fprintf(f, "%f", datum.datFloat);
168  break;
169  case vcfInfoFlag:
170  fprintf(f, "%s", datum.datString); // Flags could have values in older VCF
171  break;
172  case vcfInfoCharacter:
173  fprintf(f, "%c", datum.datChar);
174  break;
175  case vcfInfoString:
176  fprintf(f, "%s", datum.datString);
177  break;
178  default:
179  errAbort("vcfPrintDatum: Unrecognized type %d", type);
180  break;
181  }
182 }

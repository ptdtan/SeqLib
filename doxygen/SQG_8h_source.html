68  {
69  StringVector tokens = tokenizeTagValue(str);
70  if(tokens.size() != 3)

72  std::cerr << "Incorrect format for tagged value\n";

77  if(tokens[1].size() != 1 && tokens[1][0] != getTypeCode(m_value))
78  {

80  std::cerr << " Found: " << tokens[1] << "\n";

84  std::stringstream ss(tokens[2]);
